provide captions and dimensions where possible. */ -}} {{- $images := slice -}} {{- with resources.Get "media/authors/me.jpg" -}} {{- $images = $images | append (dict "@type" "ImageObject" "url" .Permalink "width" .Width "height" .Height "caption" $headshotCaption) -}} {{- end -}} {{- with resources.Get "media/authors/me-about.jpg" -}} {{- $images = $images | append (dict "@type" "ImageObject" "url" .Permalink "width" .Width "height" .Height "caption" $headshotCaption) -}} {{- end -}} {{- if fileExists "static/uploads/headshot-charite.jpg" -}} {{- $images = $images | append (dict "@type" "ImageObject" "url" (printf "%s/uploads/headshot-charite.jpg" $base) "caption" $headshotCaption) -}} {{- end -}} {{- $imageValue := "" -}} {{- if gt (len $images) 1 -}} {{- $imageValue = $images -}} {{- else if eq (len $images) 1 -}} {{- $imageValue = index $images 0 -}} {{- end -}} {{- $person := dict "@context" "https://schema.org" "@type" "Person" "@id" $personId "name" $displayName "givenName" $givenName "familyName" $familyName "honorificPrefix" "Dr" "jobTitle" $jobTitle "description" (site.Params.hugoblox.identity.description | default "") "url" site.BaseURL "image" $imageValue "sameAs" $sameAs "affiliation" $affiliations "alumniOf" $alumniOf "mainEntityOfPage" (dict "@id" site.BaseURL) -}} {{- with $author.name.pronunciation -}} {{- $person = merge $person (dict "alternateName" .) -}} {{- end -}} {{- with $email -}} {{- $person = merge $person (dict "email" .) -}} {{- end -}} {{- if $orcid -}} {{- $person = merge $person (dict "identifier" (dict "@type" "PropertyValue" "propertyID" "ORCID" "value" (strings.TrimPrefix "https://orcid.org/" $orcid) "url" $orcid ) ) -}} {{- end -}} {{- range $relMe -}} {{- end -}} {{- /* Local custom CSS */ -}}
